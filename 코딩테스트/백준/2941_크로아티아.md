```python
# 각 문자를 직접읽어서 앞뒤를 확인하는 방법은 인덱스접근에 신경써야한다
# 문자열을 찾는 방법
# dz= 와 z=이 중복되서 헷갈렸는데
# dz=은 원래 -2 해야하는데 이후에 z=검사할때에 중복되기때문에 -1만 해주면 간단하게 해결 가능하다
 
croatia = ['c=', 'c-', 'dz=', 'd-', 'lj', 'nj', 's=', 'z=']
s = input()
sum = len(s)
for i in range(0, len(croatia)):
    sum -= s.count(croatia[i])
  
print(sum)
```
